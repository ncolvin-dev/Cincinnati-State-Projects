// Name:  FindFactoral
// Abstract: Finds the factoral of a given number
// --------------------------------------------------------------------------------
int FindFactoral(int intIndex)
{
	int intFactoral = 0;
	int intRecursiveResult = 0;
	if(intIndex <= 1)
	{
		intFactoral = 1;
	}
	else
	{
		intRecursiveResult = FindFactoral(intIndex - 1);
		intFactoral = intRecursiveResult * intIndex;
	}
	
	return intFactoral;
}
